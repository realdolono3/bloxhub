async function sendMessage() { let message = document.getElementById("message").value; if (!message) { alert("Please type something!"); return; } const MAX = 1999; const webhookURL = "YOUR_WEBHOOK_URL"; while (message.length > 0) { const piece = message.slice(0, MAX); message = message.slice(MAX); // ğŸ”‘ change message value await fetch(webhookURL, { method: "POST", mode: "no-cors", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ content: piece }) // SAME LINE AS BEFORE }); await new Promise(r => setTimeout(r, 150)); } alert("Message sent!"); }
